```python
def project_evidence(bundle: EvidenceBundle) -> EvidenceProjection:
    """
    Convert an EvidenceBundle into an agent-facing EvidenceProjection.

    This is a pure adapter function: it reshapes existing evidence data
    into the summary-first projection format consumed by debate agents.
    It does NOT call text_to_sql, get_dict, or auto_debug_sql.
    """
    cards: list[EvidenceSummaryCard] = []
    uncertainty_markers: list[str] = []

    for item in bundle.items:
        # Status mapping per 01-CONTEXT.md specifics:
        # exec_status="success" + supports_conclusion=True  → "supports"
        # exec_status="success" + supports_conclusion=False → "contradicts"
        # exec_status="success" + supports_conclusion=None  → "unresolved"
        # exec_status in ("no_data","failed","field_missing") → "missing"
        if item.exec_status == "success":
            if item.supports_conclusion is True:
                status = "supports"
            elif item.supports_conclusion is False:
                status = "contradicts"
            else:
                status = "unresolved"
        else:
            status = "missing"

        card = EvidenceSummaryCard(
            card_id=f"card_{item.rule_id}",
            question=item.target,
            finding=item.result_summary,
            status=status,
            confidence=item.confidence,
            artifact_refs=[item.evidence_id],
        )
        cards.append(card)

        # Aggregate uncertainty markers
        if status == "unresolved":
            uncertainty_markers.append(
                f"[{item.rule_id}] {item.target}: 需要Agent判断"
            )
        elif status == "missing":
            uncertainty_markers.append(
                f"[{item.rule_id}] {item.target}: 证据缺失({item.exec_status})"
            )

    resolved_count = sum(1 for c in cards if c.status in ("supports", "contradicts"))
    unresolved_count = sum(1 for c in cards if c.status in ("unresolved", "missing"))

    return EvidenceProjection(
        task_header="灵活就业社保补贴资格认定",
        target_person=bundle.id_card,
        policy_scope="灵活就业社保补贴",
        cards=cards,
        uncertainty_markers=uncertainty_markers,
        total_cards=len(cards),
        resolved_count=resolved_count,
        unresolved_count=unresolved_count,
    )
```
