class TestProjectEvidence(unittest.TestCase):
    """Test project_evidence() function."""

    def setUp(self) -> None:
        self.bundle = _build_test_bundle()
        self.projection = project_evidence(self.bundle)

    def test_returns_evidence_projection_type(self) -> None:
        self.assertIsInstance(self.projection, EvidenceProjection)

    def test_task_header_is_set(self) -> None:
        self.assertEqual(self.projection.task_header, "灵活就业社保补贴资格认定")

    def test_target_person_matches_bundle_id_card(self) -> None:
        self.assertEqual(self.projection.target_person, self.bundle.id_card)

    def test_policy_scope_is_set(self) -> None:
        self.assertEqual(self.projection.policy_scope, "灵活就业社保补贴")

    def test_total_cards_matches_items_count(self) -> None:
        self.assertEqual(self.projection.total_cards, len(self.bundle.items))
        self.assertEqual(self.projection.total_cards, 5)

    def test_status_mapping_supports(self) -> None:
        """exec_status=success + supports_conclusion=True → supports"""
        card = self._find_card("card_BASIC_001")
        self.assertEqual(card.status, "supports")

    def test_status_mapping_contradicts(self) -> None:
        """exec_status=success + supports_conclusion=False → contradicts"""
        card = self._find_card("card_EXCL_001")
        self.assertEqual(card.status, "contradicts")

    def test_status_mapping_unresolved(self) -> None:
        """exec_status=success + supports_conclusion=None → unresolved"""
        card = self._find_card("card_INFER_001")
        self.assertEqual(card.status, "unresolved")

    def test_status_mapping_missing_failed(self) -> None:
        """exec_status=failed → missing"""
        card = self._find_card("card_BASIC_003")
        self.assertEqual(card.status, "missing")

    def test_status_mapping_missing_no_data(self) -> None:
        """exec_status=no_data → missing"""
        card = self._find_card("card_BASIC_004")
        self.assertEqual(card.status, "missing")

    def test_resolved_count(self) -> None:
        # supports(1) + contradicts(1) = 2
        self.assertEqual(self.projection.resolved_count, 2)

    def test_unresolved_count(self) -> None:
        # unresolved(1) + missing(2) = 3
        self.assertEqual(self.projection.unresolved_count, 3)

    def test_uncertainty_markers_present(self) -> None:
        # Should have markers for unresolved and missing items
        self.assertGreaterEqual(len(self.projection.uncertainty_markers), 3)

    def test_uncertainty_markers_contain_rule_ids(self) -> None:
        markers_text = " ".join(self.projection.uncertainty_markers)
        self.assertIn("INFER_001", markers_text)
        self.assertIn("BASIC_003", markers_text)
        self.assertIn("BASIC_004", markers_text)

    def test_card_question_maps_from_target(self) -> None:
        card = self._find_card("card_BASIC_001")
        self.assertEqual(card.question, "生存状态核查")

    def test_card_finding_maps_from_result_summary(self) -> None:
        card = self._find_card("card_BASIC_001")
        self.assertEqual(card.finding, "该人员生存状态为'生存'，数据有效。")

    def test_card_artifact_refs_contain_evidence_id(self) -> None:
        card = self._find_card("card_BASIC_001")
        self.assertIn("BASIC_001_000D", card.artifact_refs)

    def test_empty_bundle_produces_empty_projection(self) -> None:
        empty_bundle = EvidenceBundle(id_card="empty_test", items=[])
        projection = project_evidence(empty_bundle)
        self.assertEqual(projection.total_cards, 0)

    def _find_card(self, card_id: str) -> EvidenceSummaryCard:
        for card in self.projection.cards:
            if card.card_id == card_id:
                return card
        self.fail(f"Card {card_id} not found in projection")

class TestPromptTokenComparison(unittest.TestCase):
    """Verify new format is no larger than old format."""

    def test_projection_format_not_larger_than_bundle_format(self) -> None:
        bundle = _build_test_bundle()
        old_text = format_evidence_bundle(bundle)
        projection = project_evidence(bundle)
        new_text = format_projection(projection)
        # Token count approximation: character count as proxy
        # New format should not exceed old format significantly
        # (we allow up to 1.5x to account for added structure like task_header)
        self.assertLessEqual(
            len(new_text),
            len(old_text) * 1.5,
            f"New format ({len(new_text)} chars) exceeds 1.5x old format ({len(old_text)} chars)",
        )


class TestToolIsolation(unittest.TestCase):
    """Verify project_evidence() does not call forbidden tools."""

    def test_source_code_does_not_reference_text_to_sql(self) -> None:
        source = inspect.getsource(project_evidence)
        self.assertNotIn("text_to_sql", source)

    def test_source_code_does_not_reference_get_dict(self) -> None:
        source = inspect.getsource(project_evidence)
        self.assertNotIn("get_dict", source)

    def test_source_code_does_not_reference_auto_debug_sql(self) -> None:
        source = inspect.getsource(project_evidence)
        self.assertNotIn("auto_debug_sql", source)
